function displayVideos(videos) { const largeVideo = document.getElementById('large-video'); const thumbnailsContainer = document.getElementById('video-thumbnails'); if (videos.length === 0) { console.error('No videos found.'); return; } // Set first video as default setLargeVideo(videos[0].id); videos.forEach((video, index) => { const videoId = video.id; const thumbnailUrl = video.thumbnail; const title = video.title; const thumbnail = document.createElement('div'); thumbnail.classList.add('thumbnail'); thumbnail.dataset.videoId = videoId; const img = document.createElement('img'); img.src = thumbnailUrl; img.alt = title; // Add title below thumbnail const titleDiv = document.createElement('div'); titleDiv.textContent = title; titleDiv.style.padding = '5px'; titleDiv.style.fontSize = '14px'; thumbnail.appendChild(img); thumbnail.appendChild(titleDiv); thumbnailsContainer.appendChild(thumbnail); thumbnail.addEventListener('click', () => { setLargeVideo(videoId); document.querySelectorAll('.thumbnail').forEach(t => t.classList.remove('active')); thumbnail.classList.add('active'); }); if (index === 0) thumbnail.classList.add('active'); }); adjustThumbnailsHeight(); }
